#define pinBotao 13
#define pinLedVermelho 2
#define tempoLedVermelho= 5000
int contagem =0;
bool estadoAnteriorBotao=0;
void setup()
{
 Serial.begin(9600);
  pinMode(pinBotao, INPUT);
  pinMode(pinLedVermelho, OUTPUT);
}

void loop()
{
  unsigned long tempoAtual= millis();
  bool estadoAtualBotao = digitalRead(pinBotao);
  
  static unsigned long tempoAnterior=0;
    static bool estadoLed=0;
  
  if(estadoAtualBotao==1)
  {
    
   if(tempoAtual - tempoAnterior >=100)
     {
    
      estadoLed =!estadoLed;
      digitalWrite(pinLedVermelho, estadoLed);
      tempoAnterior = tempoAtual;
    
   }
     }
 estadoAnteriorBotao=estadoAtualBotao;
}
